herb_ingredient_target <- candidate_genes_df %>%
  left_join(compound_target_final, by = c("Gene" = "Symbol"),relationship = "many-to-many") %>%
  dplyr::select(Herb.Name.Pinyin, Compound.Name, Gene) %>%
  dplyr:: rename(Herb = Herb.Name.Pinyin, Ingredient = Compound.Name, Target = Gene)  %>%
  distinct() %>%
  arrange(Herb,Target)

save(herb_ingredient_target,file = "../data/processed/network/02_herb_ingredient_target.RData")

2.3 Step 2: Build herb-compound-target network

herb_ingredient <- herb_ingredient_target %>%
  dplyr::select(Herb, Ingredient) %>%
  distinct()
  
ingredient_target <- herb_ingredient_target %>%
  dplyr::select(Ingredient, Target) %>%
  distinct()
  
# edges
edges <- bind_rows(
  herb_ingredient %>% 
    dplyr::rename(source = Herb, target = Ingredient) %>% 
    mutate(interaction = "herb_ingredient"),
  ingredient_target %>%
    dplyr::rename(source = Ingredient, target = Target) %>%
    mutate(interaction = "ingredient_target")
)

# nodes
nodes <- data.frame(id = unique(c(edges$source, edges$target))) %>%
  mutate(type = case_when(
    id %in% herb_ingredient$Herb        ~ "Herb",
    id %in% ingredient_target$Ingredient  ~ "Ingredient",
    id %in% ingredient_target$Target    ~ "Target",
    TRUE                              ~ "Other"
  ))

save(nodes,edges,file = "../data/processed/network/03_HIT_network.RData")

# 相关性分析：仅保留数值列（排除分组等非数值信息）
numeric_cols <- sapply(cibersort_result, is.numeric)
# 计算Spearman相关系数（适合非正态分布数据）
cor_sorce <- cor(cibersort_result[, numeric_cols], method = 'spearman')
# 计算相关系数的P值（使用WGCNA包的函数）
cor_sorce_p <- WGCNA::corPvalueStudent(cor_sorce, nrow(cibersort_result))
# 设置颜色渐变：从深蓝（负相关）到浅黄再到深红（正相关）
col2 <- colorRampPalette(c("#053061", "#2166AC", "#4393C3", "#92C5DE", "#D1E5F0",
                           "#FDDBC7", "#F4A582", "#D6604D", "#B2182B", "#67001F"))(50)
# 定义绘制相关性热图的函数：
# 上三角显示相关系数数值，下三角显示圆圈（大小表示相关性强弱）和显著性标签
corr_plot <- function() {
  corrplot(cor_sorce,
           na.label = " ",  # 缺失值用空格表示
           diag = T,  # 显示对角线（自身相关性）
           method = "number",  # 上三角用数字表示相关系数
           type = "upper",  # 仅显示上三角
           tl.pos = "lt",  # 标签位置（左和上）
           tl.col = "black",  # 标签颜色
           tl.srt = 90,  # 标签旋转90度
           tl.cex = 0.9,  # 标签字体大小
           cl.length = 8,  # 颜色图例的分段数
           cl.cex = 0.8,  # 颜色图例字体大小
           col = col2,  # 使用自定义颜色
           number.cex = 0.7)  # 相关系数数值大小

  # 下三角添加显著性标签
  corrplot(cor_sorce,
           p.mat = cor_sorce_p,  # P值矩阵
           insig = "label_sig",  # 对不显著的相关性标记
           add = TRUE,  # 在已有图上添加
           type = "lower",  # 仅显示下三角
           method = "circle",  # 用圆圈表示相关性
           col = col2,  # 颜色与上三角一致
           na.label = " ",  # 缺失值处理
           diag = FALSE,  # 不显示对角线
           tl.pos = "n",  # 不显示标签
           cl.pos = "n",  # 不显示颜色图例
           sig.level = c(.0001, .001, .01, .05),  # 显著性水平分级
           pch.cex = 0.9,  # 显著性标签大小
           pch.col = "black")  # 显著性标签颜色
}
corr_plot()  # 绘制并显示相关性热图
