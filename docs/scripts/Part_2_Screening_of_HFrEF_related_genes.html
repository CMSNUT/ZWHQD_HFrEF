1 Differentially Expressed Genes (DEGs)

We performed differential expression analysis using the DESeq2 package1. We constructed a DESeqDataSet object from the normalized gene count matrix and sample phenotype information, with disease status (HFrEF vs. Normal) as the design variable. We then estimated dispersion parameters and fitted a negative binomial distribution model for each gene, followed by a generalized linear model and an empirical Bayes shrinkage of log2 fold changes (log2FC). Finally, we used Wald tests to compute P‑values and adjusted them using the Benjamini‑Hochberg method to obtain padj. Differentially expressed genes were defined as those with |log2FC| ≥ 0.585 and padj < 0.05.

2.7 Step 6: Extract DCM_HFrEF‑related modules and candidate hub genes

# Identify modules significantly associated with the trait
trait_of_interest <- "DCM_HFrEF"
trait_idx <- which(colnames(datTraits) == trait_of_interest)

cor_values <- moduleTraitCor[, trait_idx]
p_values   <- moduleTraitPvalue[, trait_idx]

# Modules with |cor| > 0.5 and p < 0.05
sig_modules <- names(cor_values)[p_values < 0.05 & abs(cor_values) > 0.5]

# Sort by absolute correlation (strongest first)
cor_abs <- abs(cor_values[sig_modules])
sig_modules_sorted <- names(sort(cor_abs, decreasing = TRUE))

# Display results
data.frame(Module = sig_modules_sorted,
           Correlation = cor_values[sig_modules_sorted],
           Abs_Correlation = cor_abs[order(cor_abs, decreasing = TRUE)])

# Print module sizes
for (mod in sig_modules_sorted) {
  mod_clean <- sub("^ME", "", mod)
  genes_in_mod <- names(moduleColors)[moduleColors == mod_clean]
  cat(mod, ":", length(genes_in_mod), "genes\n")
}

# Gene Significance (GS) for all genes
trait_vec <- datTraits[[trait_of_interest]]   # numeric vector (0/1 or continuous)
GS_all <- abs(cor(datExpr, trait_vec, use = "pairwise.complete.obs"))
names(GS_all) <- colnames(datExpr)

# Set thresholds for hub gene selection
use_dynamic_threshold <- FALSE   # set to TRUE for percentile‑based thresholds
MM_threshold <- 0.6
GS_threshold <- 0.2

# Percentiles used when dynamic threshold is TRUE (e.g., top 20%)
percentile_MM <- 0.8
percentile_GS <- 0.8

# 4. Extract candidate genes from each significant module
candidate_genes_list <- list()
plot_list <- list()

for (mod in sig_modules_sorted) {
  mod_clean <- sub("^ME", "", mod)
  inModule <- (moduleColors == mod_clean)

  # Skip if the module is too small (optional)
  if (sum(inModule) < 5) {
    cat("\n", mod, "skipped: module size < 5\n")
    next
  }

  # Module Membership (MM) 
  # Use moduleMembership matrix if available, otherwise compute
  if (exists("moduleMembership")) {
    # moduleMembership: rows = genes, cols = modules (named "MEblue", etc.)
    mod_col <- which(colnames(moduleMembership) == mod)
    MM_mod <- abs(moduleMembership[inModule, mod_col])
  } else {
    # Fallback: recompute MM using module eigengene
    eigengene_mod <- MEs[[mod]]
    expr_mod <- datExpr[, inModule, drop = FALSE]
    MM_mod <- abs(cor(expr_mod, eigengene_mod, use = "pairwise.complete.obs"))
  }
  names(MM_mod) <- colnames(datExpr)[inModule]

  # Gene Significance (GS) 
  GS_mod <- GS_all[inModule]
  names(GS_mod) <- colnames(datExpr)[inModule]

  # Determine thresholds (dynamic or fixed)
  if (use_dynamic_threshold) {
    mm_thresh <- quantile(MM_mod, probs = percentile_MM, na.rm = TRUE)
    gs_thresh <- quantile(GS_mod, probs = percentile_GS, na.rm = TRUE)
  } else {
    mm_thresh <- MM_threshold
    gs_thresh <- GS_threshold
  }

  # Identify candidate hub genes
  candidates <- which(MM_mod > mm_thresh & GS_mod > gs_thresh)
  candidate_genes <- names(MM_mod)[candidates]
  candidate_genes_list[[mod]] <- candidate_genes

  # Print summary
  cat("\n", mod, " (", mod_clean, "):\n", sep = "")
  cat("  Module size:", sum(inModule), "\n")
  cat("  Candidate genes:", length(candidate_genes), "\n")
  cat("  Thresholds used: MM >", round(mm_thresh, 3), ", GS >", round(gs_thresh, 3), "\n")

  # Scatter plot: MM vs GS
  plot_df <- data.frame(
    MM = MM_mod,
    GS = GS_mod,
    candidate = ifelse(names(MM_mod) %in% candidate_genes, "Yes", "No")
  )

  # Module‑trait correlation for annotation
  cor_val <- moduleTraitCor[mod, trait_of_interest]
  p_val <- moduleTraitPvalue[mod, trait_of_interest]

  p <- ggplot2::ggplot() +
    # Non‑candidate points (grey)
    ggplot2::geom_point(data = subset(plot_df, candidate == "No"), aes(x = MM, y = GS), size = 0.5, alpha = 0.6, color = "gray50") +
    # Candidate points (coloured by GS)
    ggplot2::geom_point(data = subset(plot_df, candidate == "Yes"), aes(x = MM, y = GS, color = GS), size = 0.5, alpha = 0.7) +
    ggplot2::scale_color_gradient(low = "#FDBF6F", high = "#E31A23", name = "|GS|") +
    ggplot2::geom_vline(xintercept = mm_thresh, linetype = "dashed", color = "blue", linewidth = 0.5) +
    ggplot2::geom_hline(yintercept = gs_thresh, linetype = "dashed", color = "blue", linewidth = 0.5) +
    ggplot2::labs(title = mod_clean,
                  x = "|Module Membership (MM)|",
                  y = "|Gene Significance (GS)|") +
    ggplot2::annotate("text", x = -Inf, y = Inf,
                      label = sprintf("cor = %.3f\np = %.2e", cor_val, p_val),
                      hjust = -0.2, vjust = 1.5, size = 1.5, family = "serif") +
    ggplot2::theme_bw() +
    ggplot2::theme(
      text = element_text(family = "serif"),
      panel.border = element_rect(color = "black", fill = NA, linewidth = 0.8),
      axis.line = element_line(color = "black"),
      axis.ticks = element_line(color = "black"),
      axis.ticks.length = unit(0.15, "cm"),
      axis.text = element_text(color = "black", size = 7),
      axis.title = element_text(color = "black", size = 8),
      plot.title = element_text(hjust = 0.5, size = 8),
      legend.position = "bottom",
      legend.title = element_text(size = 7)
    )

  plot_list[[mod]] <- p
}
