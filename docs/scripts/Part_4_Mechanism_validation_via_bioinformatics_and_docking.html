compute_auc <- function(gene, x, y) {
  roc_obj <- roc(response = y, predictor = x[, gene], quiet = TRUE, direction = "auto")
  auc_val <- round(auc(roc_obj), 3)
  return(auc_val)
}

gene_auc_df <- data.frame(

1.6.3 Boruta

set.seed(42)
folds <- createFolds(y_train, k = 5, list = TRUE)

importance_list <- list() 
decision_list <- list()   

for (i in seq_along(folds)) {
  train_idx <- folds[[i]]
  x_sub <- x_train[train_idx, ]
  y_sub <- y_train[train_idx]
  
  boruta_sub <- Boruta(x_sub, as.factor(y_sub), 
                       maxRuns = 100, pValue = 0.05, doTrace = 0)
  
  imp_stats <- attStats(boruta_sub)
  importance_list[[i]] <- imp_stats$medianImp
  names(importance_list[[i]]) <- rownames(imp_stats)
  
  decision_list[[i]] <- boruta_sub$finalDecision
  names(decision_list[[i]]) <- rownames(imp_stats)
}

all_features <- unique(unlist(lapply(importance_list, names)))
imp_matrix <- matrix(NA, nrow = length(all_features), ncol = 5,
                     dimnames = list(all_features, paste0("Fold", 1:5)))

for (i in 1:5) {
  imp_matrix[names(importance_list[[i]]), i] <- importance_list[[i]]
}

confirm_matrix <- sapply(decision_list, function(d) {
    confirmed <- names(d[d == "Confirmed"])
    return(all_features %in% confirmed)
})
confirm_freq <- rowSums(confirm_matrix)   # 频次 0-5

boruta_selected <- all_features[confirm_freq == 5]
cat("Boruta selected:", length(boruta_selected), "genes\n")

cat("Boruta selected:", length(boruta_selected), "genes\n")
cat(boruta_selected, "\n")

4.6 Step 5: matching SNPs (only identical alleles, no flipping)

match_identical <- function(eqtl_df, gwas_df) {
  eqtl_match <- eqtl_df %>%
    dplyr::select(gene_id, chr, pos, ref_eqtl = ref, alt_eqtl = alt,
                  beta_eqtl = beta, varbeta_eqtl = varbeta, MAF_eqtl = MAF,
                  pval_eqtl = pval, variant_id_eqtl = variant_id)
  gwas_match <- gwas_df %>%
    dplyr::select(chr, pos, ref_gwas = ref, alt_gwas = alt,
                  beta_gwas = beta, varbeta_gwas = varbeta, MAF_gwas = MAF,
                  pval_gwas = pval, variant_id_gwas = variant_id)
  matched_pos <- dplyr::inner_join(eqtl_match, gwas_match, by = c("chr", "pos"))
  matched_identical <- matched_pos %>%
    dplyr::filter(ref_eqtl == ref_gwas & alt_eqtl == alt_gwas) %>%
    dplyr::select(-ref_eqtl, -alt_eqtl, -ref_gwas, -alt_gwas)
  return(matched_identical)
}

evaluate_matching <- function(matched_df, original_eqtl, name) {
  cat("\n========== ", name, " matching quality ==========\n")
  total_eqtl <- nrow(original_eqtl)
  matched_n <- nrow(matched_df)
  cat("Match rate:", matched_n, "/", total_eqtl, sprintf("(%.2f%%)", 100 * matched_n / total_eqtl), "\n")
  
  snp_per_gene <- matched_df %>% dplyr::group_by(gene_id) %>% dplyr::summarise(n = dplyr::n())
  cat("SNPs per gene: median =", median(snp_per_gene$n), 
      ", mean =", round(mean(snp_per_gene$n),1),
      ", range =", min(snp_per_gene$n), "-", max(snp_per_gene$n), "\n")
  cat("Genes with ≥10 SNPs:", sum(snp_per_gene$n >= 10), "/", nrow(snp_per_gene), "\n")
  
  if (all(c("MAF_eqtl", "MAF_gwas") %in% names(matched_df))) {
    maf_cor <- cor(matched_df$MAF_eqtl, matched_df$MAF_gwas, method = "spearman", use = "complete.obs")
    cat("MAF Spearman correlation:", round(maf_cor, 4), "\n")
  }
  
  if (all(c("pval_eqtl", "pval_gwas") %in% names(matched_df))) {
    logp_eqtl <- -log10(matched_df$pval_eqtl)
    logp_gwas <- -log10(matched_df$pval_gwas)
    p_cor <- cor(logp_eqtl, logp_gwas, method = "spearman", use = "complete.obs")
    cat("-log10(P) Spearman correlation:", round(p_cor, 4), "\n")
  }
  
  original_genes <- unique(original_eqtl$gene_id)
  matched_genes <- unique(matched_df$gene_id)
  cat("Gene coverage:", length(matched_genes), "/", length(original_genes),
      sprintf("(%.2f%%)", 100 * length(matched_genes) / length(original_genes)), "\n")
}

filter_genes <- function(matched_df) {
  genes_keep <- matched_df %>%
    dplyr::group_by(gene_id) %>%
    dplyr::summarise(n = dplyr::n()) %>%
    dplyr::filter(n >= 10) %>%
    dplyr::pull(gene_id)
  filtered <- matched_df %>% dplyr::filter(gene_id %in% genes_keep)
  cat("Genes retained (≥10 SNPs):", length(genes_keep), "\n")
  cat("Total SNP‑gene pairs after filtering:", nrow(filtered), "\n")
  return(filtered)
}

4.9 Step 8: running P‑value based colocalisation in parallel

run_coloc_pval_gene <- function(sub_df, N_gwas, s_gwas, N_eqtl = 386,
                                p1 = 1e-4, p2 = 1e-4, p12 = 1e-5) {
  if (nrow(sub_df) < 10) return(NULL)
  d1 <- list(pvalues = sub_df$pval_eqtl,
             snp = sub_df$variant_id_eqtl,
             N = N_eqtl,
             type = "quant")
  d2 <- list(pvalues = sub_df$pval_gwas,
             snp = sub_df$variant_id_gwas,
             N = N_gwas,
             type = "cc",
             s = s_gwas)
  res <- tryCatch(coloc.abf(d1, d2, p1 = p1, p2 = p2, p12 = p12),
                  error = function(e) NULL)
  if (is.null(res)) return(NULL)
  data.frame(gene = unique(sub_df$gene_id),
             n_snps = res$summary["n snps"],
             PP.H0 = res$summary["PP.H0"],
             PP.H1 = res$summary["PP.H1"],
             PP.H2 = res$summary["PP.H2"],
             PP.H3 = res$summary["PP.H3"],
             PP.H4 = res$summary["PP.H4"])
}

4.10 Step 9: Parallel wrapper

run_parallel_pval <- function(matched_df, N_gwas, s_gwas, ncores = 8) {
  gene_list <- split(matched_df, matched_df$gene_id)
  cat("Total genes to analyse:", length(gene_list), "\n")
  plan(multisession, workers = ncores)
  results <- future_lapply(gene_list, function(x) {
    run_coloc_pval_gene(x, N_gwas, s_gwas)
  }, future.seed = TRUE)
  plan(sequential)
  results <- do.call(rbind, results)
  return(results)
}
